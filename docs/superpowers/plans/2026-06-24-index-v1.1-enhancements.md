```python
"""Detect dependency cycles (strongly-connected components) over internal edges.

Tarjan's SCC is linear; repo-level graphs are tiny (hundreds of nodes), so the
recursive form is safe and deterministic (inputs are sorted before traversal)."""
from __future__ import annotations

from collections.abc import Sequence

from .edges import Edge


def _adjacency(edges: Sequence[Edge]) -> dict[str, list[str]]:
    adj: dict[str, list[str]] = {}
    for e in edges:
        if e.external or e.to_repo is None:
            continue
        adj.setdefault(e.from_repo, [])
        adj.setdefault(e.to_repo, [])
        if e.to_repo not in adj[e.from_repo]:
            adj[e.from_repo].append(e.to_repo)
    return adj


def find_cycles(edges: Sequence[Edge]) -> list[tuple[str, ...]]:
    """Node sets of each dependency cycle, deterministically sorted. An SCC with
    >1 node, or a self-loop, is a cycle; a pure DAG returns []."""
    adj = _adjacency(edges)
    self_loops = {e.from_repo for e in edges
                  if not e.external and e.to_repo == e.from_repo}
    counter = [0]
    stack: list[str] = []
    on_stack: set[str] = set()
    index: dict[str, int] = {}
    low: dict[str, int] = {}
    sccs: list[list[str]] = []

    def connect(v: str) -> None:
        index[v] = low[v] = counter[0]
        counter[0] += 1
        stack.append(v)
        on_stack.add(v)
        for w in adj.get(v, []):
            if w not in index:
                connect(w)
                low[v] = min(low[v], low[w])
            elif w in on_stack:
                low[v] = min(low[v], index[w])
        if low[v] == index[v]:
            comp: list[str] = []
            while True:
                w = stack.pop()
                on_stack.discard(w)
                comp.append(w)
                if w == v:
                    break
            sccs.append(comp)

    for v in sorted(adj):
        if v not in index:
            connect(v)

    cycles = [tuple(sorted(c)) for c in sccs
              if len(c) > 1 or (len(c) == 1 and c[0] in self_loops)]
    return sorted(cycles)


def cycle_edge_keys(edges: Sequence[Edge],
                    cycles: Sequence[tuple[str, ...]]) -> set[tuple[str, str]]:
    """(from, to) of internal edges whose endpoints share a cycle."""
    member: dict[str, tuple[str, ...]] = {n: c for c in cycles for n in c}
    return {(e.from_repo, e.to_repo) for e in edges
            if not e.external and e.to_repo is not None
            and e.from_repo in member and member[e.from_repo] is member.get(e.to_repo)}
```

```python
"""Diff two `index graph --json` packs to surface dependency drift."""
from __future__ import annotations

from dataclasses import dataclass


@dataclass(frozen=True)
class GraphDiff:
    repos_added: tuple[str, ...]
    repos_removed: tuple[str, ...]
    edges_added: tuple[tuple[str, str], ...]
    edges_removed: tuple[tuple[str, str], ...]
    edges_changed: tuple[tuple[str, str, str, str], ...]

    def is_empty(self) -> bool:
        return not (self.repos_added or self.repos_removed or self.edges_added
                    or self.edges_removed or self.edges_changed)


def _repos(g: dict) -> set[str]:
    return {r["name"] for r in g.get("repos", [])}


def _edges(g: dict) -> dict[tuple[str, str], str]:
    out: dict[tuple[str, str], str] = {}
    for e in g.get("relations", []):
        if e.get("external") or e.get("to") is None:
            continue
        out[(e["from"], e["to"])] = e.get("confidence", "")
    return out


def diff_graphs(old: dict, new: dict) -> GraphDiff:
    or_, nr = _repos(old), _repos(new)
    oe, ne = _edges(old), _edges(new)
    common = set(oe) & set(ne)
    return GraphDiff(
        repos_added=tuple(sorted(nr - or_)),
        repos_removed=tuple(sorted(or_ - nr)),
        edges_added=tuple(sorted(set(ne) - set(oe))),
        edges_removed=tuple(sorted(set(oe) - set(ne))),
        edges_changed=tuple(sorted((f, t, oe[(f, t)], ne[(f, t)])
                                   for (f, t) in common if oe[(f, t)] != ne[(f, t)])),
    )


def render_diff(d: GraphDiff) -> str:
    if d.is_empty():
        return "no drift"
    lines: list[str] = []
    lines += [f"+ repo {n}" for n in d.repos_added]
    lines += [f"- repo {n}" for n in d.repos_removed]
    lines += [f"+ edge {f} -> {t}" for f, t in d.edges_added]
    lines += [f"- edge {f} -> {t}" for f, t in d.edges_removed]
    lines += [f"~ edge {f} -> {t}: {o} -> {n}" for f, t, o, n in d.edges_changed]
    return "\n".join(lines)


def diff_to_json(d: GraphDiff) -> dict:
    return {
        "repos_added": list(d.repos_added),
        "repos_removed": list(d.repos_removed),
        "edges_added": [list(e) for e in d.edges_added],
        "edges_removed": [list(e) for e in d.edges_removed],
        "edges_changed": [list(e) for e in d.edges_changed],
    }
```

```python
"""Build the public demo dashboard from a FABRICATED sample workspace.

Deterministic, self-contained, and PRIVATE-SAFE: the sample is invented here --
no real repository is ever read. Writes docs/demo/index.html."""
from __future__ import annotations

import tempfile
from pathlib import Path

from index_graph import viz
from index_graph.context.pack import to_json
from index_graph.graph.build import build_graph

# name -> (dependencies declared+imported, external imports)
SAMPLE: dict[str, tuple[tuple[str, ...], tuple[str, ...]]] = {
    "web-app": (("api-gateway", "ui-kit"), ()),
    "api-gateway": (("auth-core", "data-store", "shared-utils"), ()),
    "auth-core": (("shared-utils",), ("cryptography",)),
    "data-store": (("shared-utils",), ()),
    "ui-kit": (("shared-utils",), ()),
    "shared-utils": (("config-loader",), ()),     # cycle: shared-utils -> config-loader
    "config-loader": (("shared-utils",), ()),     #        config-loader -> shared-utils
}


def _materialize(root: Path) -> dict[str, Path]:
    paths: dict[str, Path] = {}
    for name, (deps, ext) in SAMPLE.items():
        repo = root / name
        mod = name.replace("-", "_")
        (repo / "src" / mod).mkdir(parents=True)
        dep_list = ", ".join(f'"{d}"' for d in deps)
        (repo / "pyproject.toml").write_text(
            f'[project]\nname = "{name}"\nversion = "0.1.0"\n'
            f"dependencies = [{dep_list}]\n", encoding="utf-8")
        imports = "".join(f"import {d.replace('-', '_')}\n" for d in deps)
        imports += "".join(f"import {e}\n" for e in ext)
        (repo / "src" / mod / "__init__.py").write_text(
            f'"""{name} -- sample."""\n{imports}', encoding="utf-8")
        (repo / "README.md").write_text(f"# {name}\n\nA fabricated sample repo.\n",
                                        encoding="utf-8")
        paths[name] = repo
    return paths


def render() -> tuple[str, dict]:
    with tempfile.TemporaryDirectory() as tmp:
        paths = _materialize(Path(tmp))
        graph = build_graph(paths)
        pack = to_json(graph)
        svg = viz.render_svg(viz.build_layout(pack))
        html = viz.render_html(pack, svg=svg, charts=viz.render_charts(pack))
    return html, pack


def build() -> str:
    html, _ = render()
    out = Path(__file__).resolve().parents[1] / "docs" / "demo" / "index.html"
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(html, encoding="utf-8")
    return str(out)


if __name__ == "__main__":
    print("wrote", build())
```
